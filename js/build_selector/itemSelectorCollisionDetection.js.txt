function itemSelectorCollisionDetection(event) {
  const root = event.target.closest(".itemSelector");
  if (!root || !root.state) return;

  const state = {
    prefix:     root.state.prefix     || "",
    postfix:    root.state.postfix    || "",
    items:      root.state.items      || {},
    namespaces: root.state.namespaces || {},
    selected:   [...root.querySelectorAll(".itemSelector_checkbox:checked")]
                  .map(cb => cb.value.trim())
  };

  const result = computeCollisions(state);

  const log = root.querySelector(".collisionWarning");
  log.innerHTML = "";
  log.style.display = "none";

  if (result.messages.length > 0) {
    log.innerHTML = result.messages.join("<br>");
    log.style.display = "block";
  }

  const copyBtn   = root.querySelector("button[name='copy']");
  const deleteBtn = root.querySelector("button[name='delete']");
  const loadBtn   = root.querySelector("button[name='load']");

  const hasNamespace = result.totalNamespaceCollisions > 0;

  if (hasNamespace) {
    copyBtn.disabled   = true;
    deleteBtn.disabled = true;
    loadBtn.disabled   = true;
  } else {
    copyBtn.disabled = (
      result.totalEmptyCollisions > 0 ||
      result.totalNameCollisions > 0
    );
    deleteBtn.disabled = false;
    loadBtn.disabled   = false;
  }
}
