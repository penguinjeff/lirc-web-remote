function build_remote(db_in, activity) {
const db=JSON.parse(JSON.stringify(db_in));

  const remote_dom = document.createElement("div");
  remote_dom.className = "container";

  // console.log(activity);
  let {remotes=[],display_name=""} = activity;

  if(display_name===""){return remote_dom;}

  let catchalls=[];
 
  const remote_list = [...remotes,""]
  const display = db["displays"][display_name]||get_default_display();

  if(remote_list.length===0){
    let dom=document.createElement("div");
    dom.innerHTML=JSON.stringify(activity)
    return dom;
  }
  // console.log("remotes");
  let unused_buttons = remote_list.map((remote_name) => {
    if(remote_name!==""){
      return JSON.parse(JSON.stringify(db.remotes[remote_name]));
    }else{
      return [];
    }
  });

  for (const row of display) {
    const row_dom = document.createElement("div");
    row_dom.className = "container-1";

    for (const column of row) {
      const column_dom = document.createElement("div");
      column_dom.className = "container-2";

      for (const module_pair of column) {
		let [module_name,remote_index=(remote_list.length-1),extra=""]=module_pair;
        const state = {
          module_name: module_name,
          remote_name: remote_list[remote_index],
		  remote_buttons: db.remotes[remote_list[remote_index]],
          unused_buttons: unused_buttons[remote_index],
          dom: document.createElement("div"),
		  extra:extra,
        };
        // console.log("all_modules")
        // console.log(db_in.all_modules);
        if(!db.all_modules[module_name]){
          console.log("Module "+module_name+" not found")
          continue;
        }
        state.dom.className = db.all_modules[module_name].class;
        if(module_name==="catchall"){
		  catchalls.push(state);
		  row_dom.appendChild(state.dom);
		  break;
		}
        // If module builds successfully, append and stop trying other modules
        if (build_module(db, state)) {
          column_dom.appendChild(state.dom);
          break;
        }
      }

      row_dom.appendChild(column_dom);
    }

    remote_dom.appendChild(row_dom);
  }
  catchalls.forEach(state=>{
    console.log(activity);
    console.log("state");
    console.log(state);
    build_catchall(state);
  });

  return remote_dom;
}
