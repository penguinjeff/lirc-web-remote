let held = false;
let holding = false;
let mousedown = false;
let holdtimer = 0;

function presshold(event)
{
  const item = event.currentTarget;
  const full = item.getAttribute("name") || "";

  const idx = full.indexOf(" ");

  let remote = "";
  let button = full;

  if (idx !== -1) {
    remote = full.slice(0, idx);
    button = full.slice(idx + 1);
  }

  function hold(remote, button)
  {
    if (holding)
    {
      held = true;
      // loop an unreasonable amount of times 1000 is unreasonable and should be cut off long before reached
      execute('macro', [[remote, button, 0, 1000]]);
      console.log("remote:" + remote + " button:" + button);
    }
  }

  console.log(remote + "+" + button);

  item.setAttribute('class', 'button_clicked');
  mousedown = true;
  holding = true;
  held = false;

  holdtimer = setTimeout(function () { hold(remote, button); }, 700);

  navigator.vibrate(700);

  console.log("remote:" + remote + " button:" + button);
  execute('macro', [[remote, button, 0, 1]]);
}


let altholdtimer = 0;

function release(event)
{
  if (!event || !event.currentTarget) {
    return;   // nothing to release
  }

  const item = event.currentTarget;
  const full = item.getAttribute("name") || "";

  const idx = full.indexOf(" ");

  let remote = "";
  let button = full;

  if (idx !== -1) {
    remote = full.slice(0, idx);
    button = full.slice(idx + 1);
  }

  function delayed_release(item)
  {
    item.className = "button";
  }

  // pointerleave often has no event or no button info
  holding = false;
  clearTimeout(holdtimer);

  if (held) {
    execute('stop', []);
  }

  setTimeout(() => delayed_release(item), 100);

  mousedown = false;
  held = false;
}


function build_button(remote = "", button_name = "", button_display = "", title = button_display) {
    function decodeHtmlEntities(str) {
        const tempElement = document.createElement('div');
        tempElement.innerHTML = str;
        return tempElement.textContent || tempElement.innerText;
    }

    if (button_name === "") {
        const dom = document.createElement('div');
        dom.className = "empty";
        return dom;
    }

    // Build inline event strings

    const dom = document.createElement('button');
    dom.className = "button";
    dom.setAttribute("name", remote + " " + button_name);

    // Tooltip
    dom.title = decodeHtmlEntities(title);

    // Inline event attributes (string form)

    dom.onpointerdown  = presshold;
    dom.onpointerup    = release;
    dom.onpointerleave = release;

    // Visible label
    dom.innerHTML = button_display;

    return dom;
}
