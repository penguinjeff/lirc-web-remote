function build_button(remote = "", button_name = "", button_display = "",title=button_display) {
    function decodeHtmlEntities(str) {
        const tempElement = document.createElement('div');
        tempElement.innerHTML = str;
        return tempElement.textContent || tempElement.innerText;
    }

    if (button_name === "") {
        const dom = document.createElement('div');
        dom.className = "empty";
        return dom;
    }
    const common = "('" + remote + "','" + button_name + "',this,event)";
    const presshold = presshold;
    const release = "release" + common;

    const dom = document.createElement('button');
    dom.className = "button";
    dom.setAttribute("name",remote+" "+button_name);

    // Tooltip text (full display text)
    dom.title = decodeHtmlEntities(title);

    // Event handlers
    dom.onpointerdown=presshold;
    dom.onpointerup=release;
    dom.onpointerleave=release;
    dom.setAttribute("title", title)


    // Visible label
    dom.innerHTML = button_display;

    return dom;
}
