window.build_catchall = function(state) {

function getDisplayName(rawName) {
  if (rawName.startsWith("KEY_")) return rawName.substring(4);
  if (rawName.startsWith("BTN_")) return rawName.substring(4);
  return rawName;
}
  let display_mode=state.display_mode;
  if(display_mode==="edit"){display_mode="none"};

  let button_list=[];
  const {remote_name,dom,unused_buttons,remote_buttons,extra={}} = state;
  const {button_columns=5,button_rows=3,alldef=""}=extra;
  if(alldef==="unused"){
    button_list=unused_buttons;
  }else{
    button_list=remote_buttons;
  }
  dom.innerHTML = "";

  // Build columns
  const cols = [];
  let index = 0;

  const pairs = button_list
  .map(name => [name, getDisplayName(name)])   // build pairs
  .sort((a, b) => a[1].localeCompare(b[1], undefined, { numeric: true })); // sort by label


  while (index < pairs.length) {
    const col = [];
    for (let r = 0; r < button_rows; r++) {
      if (index < pairs.length) {
        col.push(pairs[index++]);
      }
	  else{
		col.push(["",""]);
	  }
    }
    cols.push(col);
  }

  if(cols.length > button_columns){
    // copy button_columns-1 coloumns from the end to the beggining
    const count = button_columns - 1;

    // Copy (not remove) the last N columns
    const toCopy = cols.slice(-count);

    // Prepend them to the beginning
    cols.unshift(...toCopy);	 
  }

  const grid=document.createElement('div');
  grid.className = "container-2";
  grid.setAttribute("name", "grid");
  cols.forEach((col,i) => {
    const col_dom = document.createElement('div');
	col_dom.setAttribute("name", "Column");
	col_dom.classList.add("onewide")
    if(!(cols.length <= button_columns || ( i > (button_columns-2) && i < (2*button_columns-1)))){
	  col_dom.classList.add("hidden")
	}
	col.forEach(name => {
	  const cb=document.createElement('div');
	  cb.className="container-2";
	  let button_args={
	    remote_name:remote_name,
		button_name:name[0],
		button_display:name[1],
        display_mode:display_mode
	  }
      const btn = build_button(button_args);
	  cb.appendChild(btn);
	  col_dom.appendChild(cb);
    });
	grid.appendChild(col_dom);
  });

  dom.appendChild(grid);

  if(cols.length > button_columns){
    // Add navigator using this specific offset object
    const nav = document.createElement("div");
    nav.classList.add("container-2");
    nav.setAttribute("name","Navigator")
    const left = document.createElement("button");
    left.classList.add("button");
	left.setAttribute("name","Left")
    left.textContent = "<";

    left.onclick = catchall_move;

    const right = document.createElement("button");
    right.classList.add("button");
	right.setAttribute("name","Right")
    right.textContent = ">";

    right.onclick = catchall_move;

    nav.appendChild(left);
    nav.appendChild(right);
    dom.appendChild(nav);
  }

};
