window.catchall_wrap_toggle = function(event) {
  const toggleBtn = event.target
  const root = event.target.closest('.CatchAll');
  if (!root) { return; }
  root.wrap=!root.wrap;
  const nav = root.querySelector('[name="Navigator"]');
  const buttons = [...nav.querySelectorAll("button")]
    .filter(b => b.onclick !== event.currentTarget.onclick);

  // Now split them by reverse flag
  const reverseBtn = buttons.find(b => b.reverse === true);
  const forwardBtn = buttons.find(b => b.reverse !== true);
  forwardBtn.disabled = false;
  reverseBtn.disabled = false;
  toggleBtn.textContent = root.wraptxt[1];
  if(!root.wrap){
    toggleBtn.textContent = root.wraptxt[0];
    // All elements inside the component
    const allElements = root.querySelectorAll(root.unit);

    // Visible window (contiguous; guaranteed by the builder)
    // allElements is a NodeList, not a real array. Spreading it creates a true array,
    // which allows us to use array methods like slice(), filter(), and indexing safely.
    // Spread NodeList → Array (needed for slice/filter)
    const visible_window = [...allElements].filter(element =>
      element.style.display !== "none"
    );

    const hidden_edge  = visible_window[0].previousElementSibling;
	let hiddenIndex=-1
    if(hidden_edge){hiddenIndex = [...allElements].findIndex(el => el === hidden_edge);}
	if(hiddenIndex < visible_window.length - 1) {
      reverseBtn.disabled = true;
      const hidden_window = [...allElements].slice(0, visible_window.length-1);
      const make_visible_window = [...allElements].slice(visible_window.length, (2*visible_window.length)-1);
	  const visible=visible_window[0].style.display;
	  hidden_window.forEach(el => el.style.display = "none");
	  make_visible_window.forEach(el => el.style.display = visible);
	}
	if(hiddenIndex > (allElements.length-visible_window.length)-2) {
	  forwardBtn.disabled = true;
	}
	return;
  }
}

window.catchall_move = function(event) {

  // **************************
  // * Helper: Swap display values between two elements or element lists
  // **************************

  function swapDisplay(first, second) {
      [first.style.display,second.style.display]=[second.style.display,first.style.display]
  }

  function swapDisplays(first, second) {
      first.forEach((el,i) => swapDisplay(el,second[i]));
  }


  // **************************
  // * End of helper functions
  // **************************

  const root = event.target.closest('.CatchAll');
  if (!root) { return; }

  const isReverse = !!event.target.reverse;

  // All elements inside the component
  const allElements = root.querySelectorAll(root.unit);

  // Visible window (contiguous; guaranteed by the builder)
  // allElements is a NodeList, not a real array. Spreading it creates a true array,
  // which allows us to use array methods like slice(), filter(), and indexing safely.
  // Spread NodeList → Array (needed for slice/filter)
  const visible_window = [...allElements].filter(element =>
    element.style.display !== "none"
  );

  const first_visible = visible_window[0];
  const last_visible  = visible_window[visible_window.length - 1];

  // ***************************
  // * Determine sliding edges
  // ***************************

  // Default: moving forward
  let hidden_edge  = last_visible.nextElementSibling;
  let visible_edge = first_visible;

  // Reverse direction
  if (isReverse) {
    hidden_edge  = first_visible.previousElementSibling;
    visible_edge = last_visible;
  }

  // ***************************
  // * Case 1: Normal slide (no wrap)
  // ***************************
  if (hidden_edge) {
    swapDisplay(hidden_edge, visible_edge);

    if (!root.wrap) {
      const nav = root.querySelector('[name="Navigator"]');
      const buttons = [...nav.querySelectorAll("button")]
        .filter(b => b.onclick === event.currentTarget.onclick);

      // Now split them by reverse flag
      const reverseBtn = buttons.find(b => b.reverse === true);
      const forwardBtn = buttons.find(b => b.reverse !== true);
      forwardBtn.disabled = false;
      reverseBtn.disabled = false;
	  
	  // Compute index of the hidden edge
      const hiddenIndex = [...allElements].findIndex(el => el === hidden_edge);
	  // console.log("hiddenIndex:", hiddenIndex);
      // console.log("allElements.length:", allElements.length);
      // console.log("visible_window.length:", visible_window.length);

      if (isReverse) {
        // moving left
		// because we do not want wrapping we do not want to see the first visible_window.length-1 
		// elements so if we are headed that direction we want to disable the button when it is pressed just before that.
        if (hiddenIndex<visible_window.length) {
          reverseBtn.disabled = true;
        }
      } else {
        // moving right
		// because index starts at 0 we subtract 1 because we moved one last_visible is one more we subtract another 1 thus subtract 2 total
        if (hiddenIndex>allElements.length-2) {
          forwardBtn.disabled = true;
        }
      }	  
    }

    return;
  }
  // ***************************
  // * Case 2: Wrap slide
  // ***************************

  let start=0;
  if (isReverse) {
    start=allElements.length - visible_window.length;
  }
  const hidden_window = [...allElements].slice(start, start + visible_window.length);

  swapDisplays(hidden_window, visible_window);
};
