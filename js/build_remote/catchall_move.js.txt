window.catchall_move = function(event) {
  const item = event.currentTarget;   // ← the button that was clicked
  const direction=item.getAttribute("name")
  //get to Navigator
  let parent = item.parentElement;
  while (parent && parent.getAttribute("name")!=="Navigator") {
    parent = parent.parentElement;
  }
  if (!parent) {
    console.error("missing Navigator");
    return;
  }

  //get to outside wrapper of columns
  const wrapper = parent.previousElementSibling;

  //get to first column
  let first_column = wrapper.firstElementChild;
  while (first_column && first_column.getAttribute("name")!=="Column") {
    first_column = first_column.firstElementChild;
  }

  const columns_parent = first_column.parentElement;
  const first_visable = columns_parent.querySelector(".onewide:not(.hidden)");

  const button_columns = columns_parent.querySelectorAll(".onewide:not(.hidden)").length;

  if (!first_visable) {
    console.error("no visible columns found");
    return;
  }

  // Last visible column
  let last_visable = first_visable;
  for (let i = 1; i < button_columns; i++) {
    last_visable = last_visable.nextElementSibling;
  }
  
/***************************
 * Found all refernces onto the logic
 ***************************/  

  let hidden_edge;
  let visable_edge;
  let wrapping_edge;
  // LEFT movement
  if (direction === "Left") {
    hidden_edge = first_visable.previousElementSibling;
    visable_edge = last_visable;
    wrapping_edge = columns_parent.lastElementChild
  // RIGHT movement
  } else {
    hidden_edge = last_visable.nextElementSibling;
    visable_edge = first_visable;
    wrapping_edge = columns_parent.firstElementChild
  }
  if (hidden_edge) {

    hidden_edge.classList.remove("hidden");
    visable_edge.classList.add("hidden");

  } else {
    let current_hide = first_visable;
    for (let i = 0; i < button_columns; i++) {

      wrapping_edge.classList.remove("hidden");
      current_hide.classList.add("hidden");

      current_hide = current_hide.nextElementSibling;
      if (direction === "Left") {
        wrapping_edge = wrapping_edge.previousElementSibling;
      } else {
        wrapping_edge = wrapping_edge.nextElementSibling;
      }
    }
  }
};
