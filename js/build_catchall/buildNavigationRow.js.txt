// js/build_catchall/buildNavigationRow.js
window.catchall_move = function(item, direction, button_columns) {
  const p  = item.parentElement;          // container-2
  const gp  = p.parentElement;          // container-2
  const ps = p.previousElementSibling;    // the .onewide row
  const fc=ps.firstElementChild
  // First visible column
  const f = fc.querySelector(".onewide");

  // Last visible column
  let l = f;
  for (let i = 1; i < button_columns; i++) {
    l = l.nextElementSibling;
  }

  // Hidden neighbors
  const pref  = f.previousElementSibling; // hidden before visible
  const postl = l.nextElementSibling;     // hidden after visible

  if (direction === "l") {
    if (pref) {
//      console.log("pref:", pref.className);

      // Make pref visible
      pref.className = "onewide";

      // Hide the last visible column
      l.className = "hidden";
    } else {
 //     console.log("already at first");
	  
      // 1. Hide all current visible items
      let cur = f;
      for (let i = 0; i < button_columns; i++) {
        cur.className = "hidden";
        cur = cur.nextElementSibling;
      }

      // 2. Find the LAST element
      let last = fc.lastElementChild;

      // 3. Walk backwards button_columns - 1 times to find the start of the last block
      let start = last;
      for (let i = 0; i < button_columns; i++) {
	    start.className="onewide"
        start = start.previousElementSibling;
      }
    }
  } else {
    if (postl) {
 //     console.log("postl:", postl.className);

      // Make postl visible
      postl.className = "onewide";

      // Hide the first visible column
      f.className = "hidden";
    } else {
 //     console.log("already at last");
      // 1. Hide all current visible items
      let cur = f;
      for (let i = 0; i < button_columns; i++) {
        cur.className = "hidden";
        cur = cur.nextElementSibling;
      }
      // 2. Find the LAST element
      let first = fc.firstElementChild;

      // 3. Walk button_columns - 1 times to find the start of the last block
      let start = first;
      for (let i = 0; i < button_columns; i++) {
	    start.className="onewide"
        start = start.nextElementSibling;
      }
    }
  }
};
