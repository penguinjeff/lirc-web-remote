function build_edit(db,item) {
  console.log("build_edit")
  console.log(item);
  switch(item)
  {
    case "Activities":
	  build_activities(db);
	break;
    case "Macros":
	  build_macros(db);
	break;
    case "Displays":
	  build_displays(db);
	break;
    case "Modules":
	  build_modules(db);
	break;
    default:
	  console.log("Unkown edit "+item)
  }
  let dom=document.createElement("div");
  dom.innerHTML=JSON.stringify(item)
  return dom;
}
